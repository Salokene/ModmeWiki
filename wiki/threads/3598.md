function sound_when_i_downed()
{
    level flag::wait_till( &quot;all_players_connected&quot; );
    level flag::wait_till( &quot;initial_blackscreen_passed&quot; );
    
    wait 0.05;
    
    level endon( &quot;end_game&quot; );   
    self endon( &quot;disconnect&quot; );

    while(1)
    {
        self waittill( &quot;player_downed&quot; );
        
        self PlaySound( &quot;YOUR_SOUND&quot; );
        
        wait( 1 );
    }
