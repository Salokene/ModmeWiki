function triggers()
{       
    trigs = GetEntArray(&quot;xxx&quot;,&quot;targetname&quot;);
    foreach(trig in trigs)
    {
        trig SetCursorHint(&quot;HINT_NOICON&quot;);
        trig UseTriggerRequireLookAt();     
        trig SetHintString(&quot;If zombies don&#39;t spawn, press ^3&amp;&amp;1^7 to respawn them&quot;);

        trig thread action(trig);                   
    }           
}

{
